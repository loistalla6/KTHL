import java.lang.Math; // Importation de la bibliothèque Math

 class Complexe {
    private double re; // Attributs privés pour la partie réelle
    private double im; // Attributs privés pour la partie imaginaire

    public Complexe(double re, double im) { // Constructeur
        this.re = re;
        this.im = im;
    }

    // Getters
    public double getRe() {
        return re;
    }

    public double getIm() {
        return im;
    }

    // Méthode pour calculer le module
    public double module() {
        return Math.sqrt(re * re + im * im);
    }

    // Méthode pour calculer l'argument
    public double argument() {
        return Math.atan2(im, re);
    }

    // Méthode pour calculer le conjugué
    public Complexe conjuguer() {
        return new Complexe(re, -im);
    }

    // Addition de deux nombres complexes
    public Complexe add(Complexe x) {
        return new Complexe(this.re + x.re, this.im + x.im);
    }

    // Soustraction de deux nombres complexes
    public Complexe subtract(Complexe x) {
        return new Complexe(this.re - x.re, this.im - x.im);
    }

    // Multiplication de deux nombres complexes
    public Complexe multiply(Complexe x) {
        return new Complexe(
                this.re * x.re - this.im * x.im,
                this.re * x.im + this.im * x.re
        );
    }

    // Division de deux nombres complexes
    public Complexe divide(Complexe x) {
        double denominator = x.re * x.re + x.im * x.im;
        return new Complexe(
                (this.re * x.re + this.im * x.im) / denominator,
                (this.im * x.re - this.re * x.im) / denominator
        );
    }

    // Méthode toString pour afficher le nombre complexe
    @Override
    public String toString() {
        return "(" + re + " + " + im + "i)";
    }

    public static void main(String[] args) {
        System.out.println("Hello and welcome!");


        Complexe c1 = new Complexe(2, 3);
        Complexe c2 = new Complexe(1, -2);

        System.out.println("c1: " + c1);
        System.out.println("c2: " + c2);


        System.out.println("c1 + c2: " + c1.add(c2));
        System.out.println("c1 - c2: " + c1.subtract(c2));
        System.out.println("c1 * c2: " + c1.multiply(c2));
        System.out.println("c1 / c2: " + c1.divide(c2));


        System.out.println("Module de c1: " + c1.module());


        System.out.println("Argument de c1: " + c1.argument());


        System.out.println("Conjugué de c1: " + c1.conjuguer());
    }
}
